		cout<<a[i];

	{
		key = a[low];
		i = low;
		j = high;
		while(i < j)
		{
			for( i = i + 1; i < high; i++)
				if( a[i] > key)
					break;
			for( j = j; j > low; j--)
				if( a[j] < key)
					break;
			if(i < j)
			{	temp = a[i];
				a[i] = a[j];
				a[j] = temp;
			}
		}
		temp = a[low];
		a[low] = a[j];
		a[j] = temp;
		//cout<<"j:"<<j<<"   i:"<<i<<endl;
		quicksort(a,low,j-1);
		quicksort(a,i,high);
	}
}
